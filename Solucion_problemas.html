Se agrega una grfica de dispersin de las calificaciones reales vs las estimadas por el modelo en los datos de prueba.

In[23]:
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error, r2_score
import statsmodels.api as sm

x=data.drop(columns=["G3"]) 
y=data["G3"]  
x_train,x_test,y_train,y_test=train_test_split(x,y,test_size=0.2,random_state=42)

model = LinearRegression()
model.fit(x_train, y_train)

# Predicciones en el conjunto de prueba
y_pred = model.predict(x_test)

# Evaluacin del modelo
mse = mean_squared_error(y_test, y_pred)
r2 = r2_score(y_test, y_pred)
print("MSE:", mse)
print("R^2:", r2)

SNS.scatterplot(x=y_test, y=y_pred)
plot.xlabel("Calificacin Real")
plot.ylabel("Calificacin Prediccin")
plot.title("Comparacin de Calificaciones")
plot.grid(True)
plot.show()
